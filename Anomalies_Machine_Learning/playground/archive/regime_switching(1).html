class BacktestParameters:
    
    def __init__(self, stk_symbols, start, end, index_security, lookback_alpha, lev):
        #the stock symbols
        self.stk_symbols = stk_symbols
        #start of the data series
        self.start = start
        #end of the data series
        self.end = end
        #index security
        self.index_security = index_security
        #the lookback period for calculating alpha
        self.lookback = lookback_alpha
        self.lev = lev

class Data(BacktestParameters):
    
    def __init__(self, bp):
        
        self.stk_symbols = bp.stk_symbols
        self.start = bp.start
        self.end = bp.end
        self.index_security = bp.index_security
        self.lookback = bp.lookback
        
    def get_data(self):
        close = pd.DataFrame()

        for i in self.stk_symbols:

            df = yf.download(i, start=self.start, end=self.end, progress=False)
            df = pd.DataFrame(df['Adj Close'])
            df.columns = [i]
            if close.empty:
                close = df.copy()
            else:
                close = pd.concat([close, df], axis = 1)

        #get the daily returns
        returns = close.pct_change()[1:]

        print(returns.head())
        
        return returns
    
    def convert_monthly(self, returns):
        returns = returns.resample('M').agg(lambda x: (x + 1).prod(skipna = True) - 1)
        for i in returns.columns:
            found = False
            for date, row in returns.loc[:,i].iteritems():
                if row == 0.0 and ~found:
                    returns.loc[date, i] = np.nan
                else:
                    found = True
        print(returns)
        return returns
    
    
    
    def get_Alpha(self, returns):
        #for calculating alpha we need to only iterate through t+lookback to end. Also need to remove the index security
        df_alpha = returns[self.lookback:].drop(labels = [index_security ], axis = 1)
        #initialize the idiosyncratic volatility as well
        idiosyncratic_vol = returns[self.lookback:].drop(labels = [index_security ], axis = 1)
        

        count = 0
        #iterate through the monthly data
        for ix, row in returns.iterrows(): 
            #check to make sure we have at least the lookback
            if count >= self.lookback:
                """
                note that python will not return the last part of the index. We want to by inclusive of this
                since this is an insample measure
                """
                analysis = returns[count-self.lookback + 1:count + 1]

                for ticker in returns.columns:
                    if ticker != self.index_security :
                        #the dependent variable is our security
                        try:
                            y = analysis[ticker]
                            # the index is your independent variable
                            x = analysis[index_security ]
                            #add a constant so we can get the alpha
                            x = sm.add_constant(x)

                            #use simple ols
                            mod = sm.OLS(y, x, missing='drop')
                            res = mod.fit()
                            #get the alpha
                            alpha = res.params['const']
                            df_alpha.loc[ix, ticker] = alpha
                            #df_alpha.loc[ix, ticker] = (analysis[ticker] - analysis[index_security ]).mean()
                            idiosyncratic_vol.loc[ix, ticker] = res.resid.std()
                        except:
                            df_alpha.loc[ix, ticker] = np.nan
                            idiosyncratic_vol.loc[ix, ticker] = np.nan
                            



            count = count + 1
            
        df_alpha = df_alpha.dropna(axis = 0, how = 'all')
        idiosyncratic_vol = idiosyncratic_vol.dropna(axis = 0, how = 'all')
            
        return df_alpha, idiosyncratic_vol
    
    def get_binary(self, df_alpha, center = 0.0):
        
        binary_alpha = df_alpha.astype(float)
        binary_alpha[binary_alpha >= center] = 1.0
        binary_alpha[binary_alpha < center] = 0.0
        
        return binary_alpha
        
    
class Backtest():
    
    def __init__(self, bp, returns, mode):

    def equal_portfolio_results(self, backtest):
        
        port = pd.DataFrame(backtest.mean(axis = 1), columns = ['Regime Portfolio'])
        port.cumsum().plot(legend = True)
        equal_weighted = pd.DataFrame(self.returns.mean(axis = 1), columns = ['equal-weighted']).loc[backtest.index]
        equal_weighted.cumsum().plot(legend = True)

        print('portfolio sharpe ratio')
        print(port.mean() / port.std())
        print('equal weighted sharpe ratio')
        print(self.returns.mean(axis = 1)[backtest.index].mean() / self.returns.mean(axis = 1)[backtest.index].std())
        
        x = equal_weighted.copy()
        y = port.copy()
        x = sm.add_constant(x)

        mod = sm.OLS(y, x)
        results = mod.fit()
        print(results.summary())
        
        
        
    def print_backtest_stats(self, backtest):
        
        beta = {}
        alpha = {}
        beta_compare = {}
        alpha_compare = {}

        for i in backtest.columns:
            x = self.returns.loc[backtest.index.values, i ]
            y = backtest.loc[:, i]
            x = sm.add_constant(x)

            mod = sm.OLS(y, x)
            results = mod.fit()
            print("Strategy results for " + str(i))
            print(results.summary())
            beta[i] = results.params[i]
            alpha[i] = results.params['const'] * 12.

            #print("Normal portfolio for " + str(i))

            x = self.returns.loc[backtest.index.values, i ]
            y = self.returns.loc[backtest.index.values, i]
            x = sm.add_constant(x)

            mod = sm.OLS(y, x)
            results = mod.fit()
            #print(results.summary())

            beta_compare[i] = results.params[i]
            alpha_compare[i] = results.params['const'] * 12.


        beta = pd.DataFrame.from_dict(beta, orient = 'index', columns = ['beta'])

        alpha = pd.DataFrame.from_dict(alpha, orient = 'index', columns = ['alpha'])

        beta_compare = pd.DataFrame.from_dict(beta_compare, orient = 'index', columns = ['beta_compare'])

        alpha_compare = pd.DataFrame.from_dict(alpha_compare, orient = 'index', columns = ['alpha_compare'])

        res = pd.concat([beta, beta_compare, alpha, alpha_compare], axis = 1)
        print(res)

    def run_backtest(self, input_classifiers, lookback_backtest, inputs, outputs,  ensemble = 'no'):
        
        ensemble = ensemble.lower()
        
        if ensemble == 'no' and len(input_classifiers) > 1:
            print("For non emsemble systems the amount of input classifiers are restricted to one. The following backtest will be changed to a ensemble system")
            ensemble = 'yes'
        
        

        classifiers = {"Nearest Neighbors" : KNeighborsClassifier(3), \
                       "Linear SVM" : SVC(),\
                        "RBF SVM": SVC(gamma=2, C=1), \
                        "Gaussian Process" : GaussianProcessClassifier(1.0 * RBF(1.0)), \
                        "Decision Tree" : DecisionTreeClassifier(max_depth=5), \
                        "Random Forest" : RandomForestClassifier(max_depth=3, n_estimators=2, max_features=1, random_state=1), \
                        "Neural Net": MLPClassifier(solver='adam', learning_rate = 'adaptive', alpha=1e-3, hidden_layer_sizes=(5),max_iter= 10000, random_state=1), \
                        "AdaBoost": AdaBoostClassifier(), \
                        "Naive Bayes": GaussianNB(), \
                        "QDA": QuadraticDiscriminantAnalysis()}

        ml_backtest = pd.DataFrame()

        for ticker in inputs.keys():

            count = 0

            backtest = pd.DataFrame(outputs[ticker].loc[inputs[ticker].index][lookback_backtest:].copy())
            backtest.columns = [ticker]

            for ix, row in inputs[ticker].iterrows(): 

                if count >= lookback_backtest:
                    #the analysis df does not include the current row
                    analysis = inputs[ticker][count-lookback_backtest:count]

                    #the y var is the binary returns of the ticker
                    y = outputs[ticker].loc[analysis.index][1:].copy()
                    #the x var is the returns for the ticker on the analysis index lagged by 1
                    x = inputs[ticker].loc[analysis.index].shift(1)[1:].copy()
                    #test x represents the yesterdays x value, this is for the outsample prediction on y
                    test_x = inputs[ticker].loc[analysis.index].tail(1)
                    if ensemble == 'no':
                        for name in input_classifiers:
                            clf = classifiers[name]
                            clf.fit(x.to_numpy(), y.values.ravel())
                            predict = clf.predict(x.to_numpy())
                            predict = pd.DataFrame(data = predict, index = y.index, columns = ['predict'])
                            #mse = np.sum((y.values - predict.values.T)**2)
                            outsample = clf.predict(test_x.to_numpy())[0]
                        if outsample == 1.0:
                            backtest.loc[ix, ticker]  = returns.loc[ix, ticker] * lev
                        else:
                            backtest.loc[ix, ticker]  = 0.0
                    else:
                        all_outsample = []
                        for name in input_classifiers:
                            clf = classifiers[name]
                            clf.fit(x.to_numpy(), y.values.ravel())
                            predict = clf.predict(x.to_numpy())
                            predict = pd.DataFrame(data = predict, index = y.index, columns = ['predict'])
                            #mse = np.sum((y.values - predict.values.T)**2)
                            outsample = clf.predict(test_x.to_numpy())[0]
                            all_outsample.append(outsample)

                        avg = sum(all_outsample) / len(all_outsample)     
                        if avg >= 0.5:
                            backtest.loc[ix, ticker]  = returns.loc[ix, ticker] * lev
                        else:
                            backtest.loc[ix, ticker]  = 0.0
                count = count + 1
            if ml_backtest.empty:
                ml_backtest = backtest.copy()
            else:
                ml_backtest = pd.concat([ml_backtest, backtest], axis = 1)

            
        return ml_backtest
